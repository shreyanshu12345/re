The following dummy code snippet illustrates a Redux async action used for secure API integration and state management.

\begin{tcblisting}{listing only,

    breakable,

        basicstyle=\small\ttfamily,
        keywordstyle=\color{blue},
        commentstyle=\color{gray},
        backgroundcolor=\color{gray!25},
        showstringspaces=false,
        breaklines=true
    }
}
import axios from "axios";

export const fetchCustomerData = () => {
    return async (dispatch) => {
        dispatch({ type: "FETCH_START" });

        try {
            const response = await axios.get("/api/customer");

            dispatch({
                type: "FETCH_SUCCESS",
                payload: response.data
            });
        } catch (error) {
            dispatch({
                type: "FETCH_ERROR",
                payload: error.message
            });
        }
    };
};
\end{tcblisting}

\section{Authentication Workflow Example}

The following dummy code snippet represents a simplified OTP verification workflow used for secure customer authentication.

\begin{tcblisting}{listing only,
    colback=gray!10!white, 
    breakable, 
    boxsep=0pt,
    top=1mm,
    bottom=1mm,
    left=1mm,
    right=1mm,
    listing options={
        language=JavaScript,
        basicstyle=\small\ttfamily,
        keywordstyle=\color{blue},
        commentstyle=\color{gray},
        backgroundcolor=\color{gray!25},
        showstringspaces=false,
        breaklines=true
    }
}
const verifyOTP = async (otpCode) => {
    try {
        const response = await fetch("/api/verify-otp", {
            method: "POST",
            headers: {
                "Content-Type": "application/json"
            },
            body: JSON.stringify({ otp: otpCode })
        });

        const data = await response.json();

        if (data.success) {
            console.log("Authentication successful");
        } else {
            console.log("Invalid OTP");
        }
    } catch (error) {
        console.log("Verification failed");
    }
};
\end{tcblisting}

const cloneOffer = (existingOffer) => {
    return {
        ...existingOffer,
        version: existingOffer.version + 1,
        status: "DRAFT"
    };
};

const publishOffer = (offer) => {
    if (offer.approved) {
        offer.status = "LIVE";
    }

    return offer;
